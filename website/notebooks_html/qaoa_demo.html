# Cell 3: helper to optimize angles classically (maximize expected cut)
import math

def classical_optimize(n, edges, p, maxiter=200):
    def loss(x):
        gammas = x[:p]
        betas = x[p:]
        state = qaoa_state(n, edges, gammas, betas)
        val = expected_cut(n, edges, state)
        return -val

    x0 = np.random.uniform(0, math.pi, size=2*p)
    res = minimize(loss, x0, method='Nelder-Mead', options={'maxiter': maxiter})
    sol = res.x
    return sol[:p], sol[p:]

# Cell 6: predict with GNN and evaluate
# build adjacency and node features
A = np.zeros((n,n), dtype=float)
for u,v in edges:
    A[u,v] = 1; A[v,u] = 1
A = A + np.eye(n)
deg = A.sum(axis=1)
X = deg.reshape(n,1)

with torch.no_grad():
    At = torch.tensor(A, dtype=torch.float32)
    Xt = torch.tensor(X, dtype=torch.float32)
    out = model(Xt, At)
    out_flat = out.view(-1).numpy()
    half = out_flat.shape[0]//2
    pred_g, pred_b = out_flat[:half], out_flat[half:]

state_pred = qaoa_state(n, edges, pred_g, pred_b)
val_pred = expected_cut(n, edges, state_pred)
print('GNN predicted expected cut:', val_pred)
print('\nClassical angles (gamma,beta):', opt_gammas, opt_betas)
print('GNN angles (gamma,beta):', pred_g, pred_b)
